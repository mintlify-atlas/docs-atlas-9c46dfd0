---
title: "List Routines"
api: "GET /api/v1/routines"
description: "Retrieve a paginated list of workout routines with optional filtering and sorting"
---

## Query Parameters

<ParamField query="page" type="integer" default="1">
  Page number for pagination (minimum: 1)
</ParamField>

<ParamField query="limit" type="integer" default="20">
  Number of routines per page (minimum: 1, maximum: 100)
</ParamField>

<ParamField query="q" type="string">
  Search query to filter routines by name, block title, or exercise name (case-insensitive)
</ParamField>

<ParamField query="tag" type="string">
  Comma-separated list of tags to filter by (e.g., "strength,hypertrophy")
</ParamField>

<ParamField query="templatesOnly" type="boolean">
  When true, only return routines marked as templates
</ParamField>

<ParamField query="includeArchived" type="boolean">
  When true, include archived routines in results. By default, archived routines are excluded
</ParamField>

<ParamField query="sort" type="string" default="-createdAt">
  Sort order for results. Options:
  - `name` - Sort by name A-Z
  - `-name` - Sort by name Z-A
  - `createdAt` - Sort by creation date (oldest first)
  - `-createdAt` - Sort by creation date (newest first)
  - `lastPerformedAt` - Sort by last performed date (oldest first)
  - `-lastPerformedAt` - Sort by last performed date (most recent first)
  - `timesPerformed` - Sort by times performed (ascending)
  - `-timesPerformed` - Sort by times performed (descending)
</ParamField>

## Response

<ResponseField name="page" type="integer">
  Current page number
</ResponseField>

<ResponseField name="limit" type="integer">
  Number of items per page
</ResponseField>

<ResponseField name="total" type="integer">
  Total number of routines matching the filters
</ResponseField>

<ResponseField name="items" type="array">
  Array of routine objects
  
  <ResponseField name="_id" type="string">
    Unique routine identifier
  </ResponseField>
  
  <ResponseField name="userId" type="string">
    ID of the user who owns this routine
  </ResponseField>
  
  <ResponseField name="name" type="string">
    Name of the routine
  </ResponseField>
  
  <ResponseField name="blocks" type="array">
    Array of routine blocks, each containing exercises and configuration
    
    <ResponseField name="title" type="string">
      Optional title for the block (e.g., "Lower", "Push", "Core")
    </ResponseField>
    
    <ResponseField name="position" type="integer">
      Order of the block within the routine
    </ResponseField>
    
    <ResponseField name="exercises" type="array">
      List of exercises in this block
      
      <ResponseField name="name" type="string">
        Exercise name
      </ResponseField>
      
      <ResponseField name="sets" type="integer">
        Number of sets
      </ResponseField>
      
      <ResponseField name="reps" type="string">
        Rep scheme (e.g., "5", "8-12", "AMRAP")
      </ResponseField>
      
      <ResponseField name="restSec" type="integer">
        Rest time between sets in seconds
      </ResponseField>
      
      <ResponseField name="position" type="integer">
        Order of the exercise within the block
      </ResponseField>
      
      <ResponseField name="loadKg" type="number">
        Optional weight in kilograms
      </ResponseField>
      
      <ResponseField name="rir" type="integer">
        Reps in reserve (0-10)
      </ResponseField>
      
      <ResponseField name="tempo" type="string">
        Tempo prescription (e.g., "3-1-1")
      </ResponseField>
      
      <ResponseField name="notes" type="string">
        Additional notes or instructions
      </ResponseField>
      
      <ResponseField name="videoUrl" type="string">
        URL to instructional video
      </ResponseField>
      
      <ResponseField name="cues" type="array">
        Array of coaching cues
      </ResponseField>
    </ResponseField>
    
    <ResponseField name="exerciseType" type="string">
      Type of exercises in this block: "strength", "hypertrophy", "conditioning", or "mobility"
    </ResponseField>
    
    <ResponseField name="rounds" type="integer">
      Number of rounds to complete for this block
    </ResponseField>
    
    <ResponseField name="restBetweenExercisesSec" type="integer">
      Rest time between exercises in seconds
    </ResponseField>
    
    <ResponseField name="timer" type="object">
      Timer configuration for the block
      
      <ResponseField name="mode" type="string">
        Timer mode: "countdown", "emom", "amrap", or "tabata"
      </ResponseField>
      
      <ResponseField name="seconds" type="integer">
        Duration in seconds
      </ResponseField>
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="tags" type="array">
    Array of tags associated with the routine
  </ResponseField>
  
  <ResponseField name="isTemplate" type="boolean">
    Whether this routine is marked as a template
  </ResponseField>
  
  <ResponseField name="isArchived" type="boolean">
    Whether this routine has been archived
  </ResponseField>
  
  <ResponseField name="estimatedDurationMin" type="integer">
    Estimated duration of the routine in minutes
  </ResponseField>
  
  <ResponseField name="lastPerformedAt" type="string">
    ISO 8601 date when the routine was last performed
  </ResponseField>
  
  <ResponseField name="timesPerformed" type="integer">
    Number of times this routine has been performed
  </ResponseField>
  
  <ResponseField name="createdAt" type="string">
    ISO 8601 date when the routine was created
  </ResponseField>
  
  <ResponseField name="updatedAt" type="string">
    ISO 8601 date when the routine was last updated
  </ResponseField>
</ResponseField>
