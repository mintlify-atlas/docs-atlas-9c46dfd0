---
title: "Create Routine"
api: "POST /api/v1/routines"
description: "Create a new workout routine with exercises, sets, reps, and scheduling"
---

## Body Parameters

<ParamField body="name" type="string" required>
  Name of the routine (minimum 2 characters)
</ParamField>

<ParamField body="blocks" type="array" default="[]">
  Array of routine blocks containing exercises and configuration
  
  <ParamField body="title" type="string">
    Optional title for the block (e.g., "Lower", "Push", "Core")
  </ParamField>
  
  <ParamField body="position" type="integer" required>
    Order of the block within the routine (minimum: 1)
  </ParamField>
  
  <ParamField body="exercises" type="array" default="[]">
    List of exercises in this block
    
    <ParamField body="name" type="string" required>
      Exercise name
    </ParamField>
    
    <ParamField body="sets" type="integer" required>
      Number of sets (minimum: 1)
    </ParamField>
    
    <ParamField body="reps" type="string" required>
      Rep scheme (e.g., "5", "8-12", "AMRAP")
    </ParamField>
    
    <ParamField body="restSec" type="integer" default="90">
      Rest time between sets in seconds (minimum: 0)
    </ParamField>
    
    <ParamField body="position" type="integer" required>
      Order of the exercise within the block (minimum: 1)
    </ParamField>
    
    <ParamField body="loadKg" type="number">
      Weight in kilograms (must be positive)
    </ParamField>
    
    <ParamField body="rir" type="integer">
      Reps in reserve (0-10)
    </ParamField>
    
    <ParamField body="tempo" type="string">
      Tempo prescription (e.g., "3-1-1" for 3 seconds eccentric, 1 second pause, 1 second concentric)
    </ParamField>
    
    <ParamField body="notes" type="string">
      Additional notes or instructions for the exercise
    </ParamField>
    
    <ParamField body="videoUrl" type="string">
      URL to instructional video (must be valid URL)
    </ParamField>
    
    <ParamField body="cues" type="array" default="[]">
      Array of coaching cues or reminders
    </ParamField>
  </ParamField>
  
  <ParamField body="exerciseType" type="string">
    Type of exercises in this block: "strength", "hypertrophy", "conditioning", or "mobility"
  </ParamField>
  
  <ParamField body="rounds" type="integer">
    Number of rounds to complete for this block (minimum: 1)
  </ParamField>
  
  <ParamField body="restBetweenExercisesSec" type="integer">
    Rest time between exercises in seconds (minimum: 0)
  </ParamField>
  
  <ParamField body="timer" type="object">
    Timer configuration for the block
    
    <ParamField body="mode" type="string" required>
      Timer mode: "countdown", "emom" (every minute on the minute), "amrap" (as many reps as possible), or "tabata"
    </ParamField>
    
    <ParamField body="seconds" type="integer" required>
      Duration in seconds (minimum: 1)
    </ParamField>
  </ParamField>
</ParamField>

<ParamField body="tags" type="array">
  Array of tags to categorize the routine (e.g., ["strength", "lower-body"])
</ParamField>

<ParamField body="isTemplate" type="boolean" default="false">
  Whether this routine should be marked as a template for reuse
</ParamField>

<ParamField body="isArchived" type="boolean" default="false">
  Whether this routine should be created as archived
</ParamField>

<ParamField body="estimatedDurationMin" type="integer">
  Estimated duration of the routine in minutes (minimum: 1)
</ParamField>

<ParamField body="timesPerformed" type="integer" default="0">
  Number of times this routine has been performed (minimum: 0)
</ParamField>

## Response

<ResponseField name="_id" type="string">
  Unique routine identifier
</ResponseField>

<ResponseField name="userId" type="string">
  ID of the user who owns this routine
</ResponseField>

<ResponseField name="name" type="string">
  Name of the routine
</ResponseField>

<ResponseField name="blocks" type="array">
  Array of routine blocks with exercises
</ResponseField>

<ResponseField name="tags" type="array">
  Array of tags associated with the routine
</ResponseField>

<ResponseField name="isTemplate" type="boolean">
  Whether this routine is marked as a template
</ResponseField>

<ResponseField name="isArchived" type="boolean">
  Whether this routine has been archived
</ResponseField>

<ResponseField name="estimatedDurationMin" type="integer">
  Estimated duration of the routine in minutes
</ResponseField>

<ResponseField name="timesPerformed" type="integer">
  Number of times this routine has been performed
</ResponseField>

<ResponseField name="createdAt" type="string">
  ISO 8601 date when the routine was created
</ResponseField>

<ResponseField name="updatedAt" type="string">
  ISO 8601 date when the routine was last updated
</ResponseField>
