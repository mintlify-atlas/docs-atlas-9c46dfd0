---
title: "Get Progress Entry"
api: "GET /api/v1/progress/:id"
description: "Retrieve a specific progress entry by its unique identifier"
---

Retrieve detailed information about a specific progress entry for the authenticated user. This endpoint returns all measurements, photos, and metadata associated with the entry.

## Path Parameters

<ParamField path="id" type="string" required>
  Unique identifier of the progress entry to retrieve
  
  Example: `prg_123` or MongoDB ObjectId
</ParamField>

## Response

<ResponseField name="id" type="string">
  Unique identifier for the progress entry
</ResponseField>

<ResponseField name="userId" type="string">
  ID of the user who owns this entry
</ResponseField>

<ResponseField name="date" type="string">
  Date of the progress entry (YYYY-MM-DD format)
  
  Example: `2025-11-05`
</ResponseField>

<ResponseField name="timeOfDay" type="string">
  Time of day when measurements were taken
  
  Options: `morning`, `evening`, `other`
</ResponseField>

<ResponseField name="timezone" type="string">
  Timezone identifier for the entry
</ResponseField>

<ResponseField name="unitSystem" type="string">
  Unit system used for measurements
  
  Options: `metric`, `imperial`
</ResponseField>

<ResponseField name="source" type="string">
  Source of the progress data
  
  Options: `manual`, `device`, `import`
</ResponseField>

<ResponseField name="weight_kg" type="number">
  Body weight in kilograms
  
  Example: `72.4`
</ResponseField>

<ResponseField name="waist_cm" type="number">
  Waist circumference in centimeters
  
  Example: `81`
</ResponseField>

<ResponseField name="body" type="object">
  Body composition measurements
  
  <Expandable title="body properties">
    <ResponseField name="bodyFatPct" type="number">
      Body fat percentage (1-75)
      
      Example: `18.6`
    </ResponseField>
    
    <ResponseField name="bodyFatMethod" type="string">
      Method used to measure body fat
      
      Options: `bmi`, `impedance`, `caliper`, `scan`
    </ResponseField>
    
    <ResponseField name="chest_cm" type="number">
      Chest circumference in centimeters
    </ResponseField>
    
    <ResponseField name="hip_cm" type="number">
      Hip circumference in centimeters
    </ResponseField>
    
    <ResponseField name="thigh_cm" type="number">
      Thigh circumference in centimeters
    </ResponseField>
    
    <ResponseField name="arm_cm" type="number">
      Arm circumference in centimeters
    </ResponseField>
    
    <ResponseField name="neck_cm" type="number">
      Neck circumference in centimeters
    </ResponseField>
    
    <ResponseField name="estimatedLeanMass_kg" type="number">
      Estimated lean body mass in kilograms
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="vitals" type="object">
  Vital signs measurements
  
  <Expandable title="vitals properties">
    <ResponseField name="rhr_bpm" type="number">
      Resting heart rate in beats per minute
    </ResponseField>
    
    <ResponseField name="bp_systolic" type="number">
      Systolic blood pressure
    </ResponseField>
    
    <ResponseField name="bp_diastolic" type="number">
      Diastolic blood pressure
    </ResponseField>
    
    <ResponseField name="spo2_pct" type="number">
      Blood oxygen saturation percentage (50-100)
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="wellness" type="object">
  Wellness and recovery metrics
  
  <Expandable title="wellness properties">
    <ResponseField name="sleep_hours" type="number">
      Hours of sleep (0-24)
    </ResponseField>
    
    <ResponseField name="energy_1to5" type="integer">
      Energy level rating (1-5)
    </ResponseField>
    
    <ResponseField name="stress_1to5" type="integer">
      Stress level rating (1-5)
    </ResponseField>
    
    <ResponseField name="soreness_1to5" type="integer">
      Muscle soreness rating (1-5)
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="activity" type="object">
  Activity and exercise data
  
  <Expandable title="activity properties">
    <ResponseField name="steps" type="integer">
      Number of steps taken
      
      Example: `8500`
    </ResponseField>
    
    <ResponseField name="timeTrainedMin" type="integer">
      Time spent training in minutes
    </ResponseField>
    
    <ResponseField name="vo2max" type="number">
      VO2 max measurement
    </ResponseField>
    
    <ResponseField name="workoutId" type="string">
      Reference to associated workout
    </ResponseField>
    
    <ResponseField name="caloriesBurned_est" type="integer">
      Estimated calories burned
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="notes" type="string">
  User notes for this progress entry
  
  Example: `Felt great this week.`
</ResponseField>

<ResponseField name="photos" type="array">
  Array of progress photos
  
  <Expandable title="photo properties">
    <ResponseField name="url" type="string">
      URL of the photo
    </ResponseField>
    
    <ResponseField name="publicId" type="string">
      Cloudinary public ID
    </ResponseField>
    
    <ResponseField name="pose" type="string">
      Photo pose type
      
      Options: `front`, `side`, `back`
    </ResponseField>
    
    <ResponseField name="lighting" type="string">
      Lighting conditions
      
      Options: `natural`, `artificial`, `unknown`
    </ResponseField>
    
    <ResponseField name="notes" type="string">
      Notes about the photo
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="startPhoto" type="object">
  Reference photo marking the start of progress tracking
  
  <Expandable title="photo properties">
    <ResponseField name="url" type="string">
      URL of the photo
    </ResponseField>
    
    <ResponseField name="publicId" type="string">
      Cloudinary public ID
    </ResponseField>
    
    <ResponseField name="notes" type="string">
      Notes about the photo
    </ResponseField>
    
    <ResponseField name="capturedAt" type="string">
      Timestamp when photo was captured
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="comparePhoto" type="object">
  Reference photo for comparison
  
  <Expandable title="photo properties">
    <ResponseField name="url" type="string">
      URL of the photo
    </ResponseField>
    
    <ResponseField name="publicId" type="string">
      Cloudinary public ID
    </ResponseField>
    
    <ResponseField name="notes" type="string">
      Notes about the photo
    </ResponseField>
    
    <ResponseField name="capturedAt" type="string">
      Timestamp when photo was captured
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="tags" type="array">
  Array of tag strings for categorization
</ResponseField>

<ResponseField name="createdAt" type="string">
  Timestamp when entry was created
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Timestamp when entry was last updated
</ResponseField>

<RequestExample>
```bash cURL
curl -X GET "https://api.fitboard.app/api/v1/progress/prg_123" \
  -H "Authorization: Bearer YOUR_TOKEN"
```

```javascript JavaScript
const response = await fetch('https://api.fitboard.app/api/v1/progress/prg_123', {
  method: 'GET',
  headers: {
    'Authorization': 'Bearer YOUR_TOKEN'
  }
});

const data = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://api.fitboard.app/api/v1/progress/prg_123',
    headers={'Authorization': 'Bearer YOUR_TOKEN'}
)

data = response.json()
```
</RequestExample>
