---
title: "Create Progress Entry"
api: "POST /api/v1/progress"
description: "Create a new progress entry with measurements, photos, and notes"
---

Create a new progress entry for the authenticated user. Progress entries track body measurements, photos, wellness metrics, and activity data over time.

## Body Parameters

<ParamField body="date" type="string" required>
  Date of the progress entry in YYYY-MM-DD format
  
  Example: `2025-11-05`
</ParamField>

<ParamField body="timeOfDay" type="string" default="other">
  Time of day when measurements were taken
  
  Options: `morning`, `evening`, `other`
</ParamField>

<ParamField body="timezone" type="string">
  Timezone identifier for the entry
  
  Example: `America/New_York`
</ParamField>

<ParamField body="unitSystem" type="string" default="metric">
  Unit system for measurements
  
  Options: `metric`, `imperial`
</ParamField>

<ParamField body="source" type="string" default="manual">
  Source of the progress data
  
  Options: `manual`, `device`, `import`
</ParamField>

<ParamField body="weight_kg" type="number">
  Body weight in kilograms (must be positive)
  
  Example: `72.4`
</ParamField>

<ParamField body="waist_cm" type="number">
  Waist circumference in centimeters (must be positive)
  
  Example: `81`
</ParamField>

<ParamField body="body" type="object">
  Body composition measurements
  
  <Expandable title="body properties">
    <ParamField body="body.bodyFatPct" type="number">
      Body fat percentage (1-75)
      
      Example: `18.6`
    </ParamField>
    
    <ParamField body="body.bodyFatMethod" type="string">
      Method used to measure body fat
      
      Options: `bmi`, `impedance`, `caliper`, `scan`
    </ParamField>
    
    <ParamField body="body.chest_cm" type="number">
      Chest circumference in centimeters (must be positive)
    </ParamField>
    
    <ParamField body="body.hip_cm" type="number">
      Hip circumference in centimeters (must be positive)
    </ParamField>
    
    <ParamField body="body.thigh_cm" type="number">
      Thigh circumference in centimeters (must be positive)
    </ParamField>
    
    <ParamField body="body.arm_cm" type="number">
      Arm circumference in centimeters (must be positive)
    </ParamField>
    
    <ParamField body="body.neck_cm" type="number">
      Neck circumference in centimeters (must be positive)
    </ParamField>
    
    <ParamField body="body.estimatedLeanMass_kg" type="number">
      Estimated lean body mass in kilograms (must be positive)
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="vitals" type="object">
  Vital signs measurements
  
  <Expandable title="vitals properties">
    <ParamField body="vitals.rhr_bpm" type="number">
      Resting heart rate in beats per minute (must be positive)
    </ParamField>
    
    <ParamField body="vitals.bp_systolic" type="number">
      Systolic blood pressure (must be positive)
    </ParamField>
    
    <ParamField body="vitals.bp_diastolic" type="number">
      Diastolic blood pressure (must be positive)
    </ParamField>
    
    <ParamField body="vitals.spo2_pct" type="number">
      Blood oxygen saturation percentage (50-100)
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="wellness" type="object">
  Wellness and recovery metrics
  
  <Expandable title="wellness properties">
    <ParamField body="wellness.sleep_hours" type="number">
      Hours of sleep (0-24)
      
      Example: `7.5`
    </ParamField>
    
    <ParamField body="wellness.energy_1to5" type="integer">
      Energy level rating (1-5, where 5 is highest)
    </ParamField>
    
    <ParamField body="wellness.stress_1to5" type="integer">
      Stress level rating (1-5, where 5 is highest)
    </ParamField>
    
    <ParamField body="wellness.soreness_1to5" type="integer">
      Muscle soreness rating (1-5, where 5 is most sore)
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="activity" type="object">
  Activity and exercise data
  
  <Expandable title="activity properties">
    <ParamField body="activity.steps" type="integer">
      Number of steps taken (must be non-negative)
      
      Example: `8500`
    </ParamField>
    
    <ParamField body="activity.timeTrainedMin" type="integer">
      Time spent training in minutes (must be non-negative)
    </ParamField>
    
    <ParamField body="activity.vo2max" type="number">
      VO2 max measurement (must be non-negative)
    </ParamField>
    
    <ParamField body="activity.workoutId" type="string">
      Reference to associated workout routine
    </ParamField>
    
    <ParamField body="activity.caloriesBurned_est" type="integer">
      Estimated calories burned (must be non-negative)
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="notes" type="string">
  User notes and observations for this entry
  
  Example: `Felt great this week. Weight is down slightly.`
</ParamField>

<ParamField body="photos" type="array">
  Array of progress photo objects with URLs and metadata
  
  <Expandable title="photo properties">
    <ParamField body="photos[].url" type="string" required>
      URL of the photo (must be a valid URL)
    </ParamField>
    
    <ParamField body="photos[].pose" type="string">
      Photo pose type
      
      Options: `front`, `side`, `back`
    </ParamField>
    
    <ParamField body="photos[].lighting" type="string">
      Lighting conditions when photo was taken
      
      Options: `natural`, `artificial`, `unknown`
    </ParamField>
    
    <ParamField body="photos[].notes" type="string">
      Notes about the photo
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="startPhoto" type="object">
  Reference photo marking the start of progress tracking
  
  <Expandable title="photo properties">
    <ParamField body="startPhoto.url" type="string" required>
      URL of the photo (must be a valid URL)
    </ParamField>
    
    <ParamField body="startPhoto.publicId" type="string">
      Cloudinary public ID for the photo
    </ParamField>
    
    <ParamField body="startPhoto.notes" type="string">
      Notes about the photo
    </ParamField>
    
    <ParamField body="startPhoto.capturedAt" type="string">
      Timestamp when photo was captured
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="comparePhoto" type="object">
  Reference photo for progress comparison
  
  <Expandable title="photo properties">
    <ParamField body="comparePhoto.url" type="string" required>
      URL of the photo (must be a valid URL)
    </ParamField>
    
    <ParamField body="comparePhoto.publicId" type="string">
      Cloudinary public ID for the photo
    </ParamField>
    
    <ParamField body="comparePhoto.notes" type="string">
      Notes about the photo
    </ParamField>
    
    <ParamField body="comparePhoto.capturedAt" type="string">
      Timestamp when photo was captured
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="tags" type="array">
  Array of tag strings for categorization
  
  Example: `["bulking", "strength"]`
</ParamField>
