---
title: "Get Recipe"
api: "GET /api/v1/recipes/{id}"
description: "Retrieve a specific recipe by ID"
---

Retrieves the details of a single recipe. Only recipes owned by the authenticated user can be accessed.

## Path Parameters

<ParamField path="id" type="string" required>
  The unique identifier of the recipe
</ParamField>

## Response

<ResponseField name="_id" type="string">
  Unique recipe ID
</ResponseField>

<ResponseField name="userId" type="string">
  ID of the user who owns this recipe
</ResponseField>

<ResponseField name="name" type="string">
  Recipe name
</ResponseField>

<ResponseField name="date" type="string">
  Date in YYYY-MM-DD format
</ResponseField>

<ResponseField name="servings" type="integer">
  Number of servings
</ResponseField>

<ResponseField name="ingredients" type="Ingredient[]">
  Array of ingredient objects
  
  <ResponseField name="name" type="string">
    Ingredient name
  </ResponseField>
  
  <ResponseField name="amount" type="number">
    Quantity of the ingredient
  </ResponseField>
  
  <ResponseField name="unit" type="string">
    Unit of measurement: `g`, `ml`, or `unit`
  </ResponseField>
  
  <ResponseField name="nutritionBasis" type="object">
    Nutrition information basis
    
    <ResponseField name="amount" type="number">
      Basis amount for nutrition calculation
    </ResponseField>
    
    <ResponseField name="unit" type="string">
      Basis unit: `g`, `ml`, or `unit`
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="macrosPerBasis" type="object">
    Macronutrients per nutrition basis
    
    <ResponseField name="kcal" type="number">
      Calories (kcal)
    </ResponseField>
    
    <ResponseField name="protein" type="number">
      Protein in grams
    </ResponseField>
    
    <ResponseField name="carbohydrate" type="number">
      Carbohydrates in grams
    </ResponseField>
    
    <ResponseField name="fat" type="number">
      Fat in grams
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="gramsPerUnit" type="number">
    Weight in grams per unit (optional, for unit-based ingredients)
  </ResponseField>
  
  <ResponseField name="densityGPerMl" type="number">
    Density in grams per milliliter (optional, for liquid ingredients)
  </ResponseField>
</ResponseField>

<ResponseField name="categoryIds" type="string[]">
  Array of category IDs associated with this recipe
</ResponseField>

<ResponseField name="createdAt" type="string">
  Timestamp when the recipe was created
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Timestamp when the recipe was last updated
</ResponseField>

<RequestExample>
```bash cURL
curl -X GET "https://api.fitboard.app/api/v1/recipes/65f3a2b1c4d5e6f7a8b9c0d1" \
  -H "Authorization: Bearer YOUR_TOKEN"
```

```javascript JavaScript
const response = await fetch('https://api.fitboard.app/api/v1/recipes/65f3a2b1c4d5e6f7a8b9c0d1', {
  headers: {
    'Authorization': 'Bearer YOUR_TOKEN'
  }
});

const data = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://api.fitboard.app/api/v1/recipes/65f3a2b1c4d5e6f7a8b9c0d1',
    headers={'Authorization': 'Bearer YOUR_TOKEN'}
)

data = response.json()
```
</RequestExample>
