---
title: "Update Recipe"
api: "PATCH /api/v1/recipes/{id}"
description: "Partially update an existing recipe"
---

Updates specific fields of an existing recipe. Only the fields provided in the request body will be updated. All fields are optional.

## Path Parameters

<ParamField path="id" type="string" required>
  The unique identifier of the recipe to update
</ParamField>

## Body Parameters

All fields are optional. Only include the fields you want to update.

<ParamField body="name" type="string">
  Recipe name (minimum 2 characters). Must be unique for the user.
</ParamField>

<ParamField body="date" type="string">
  Date in YYYY-MM-DD format (e.g., "2024-03-15")
</ParamField>

<ParamField body="servings" type="integer">
  Number of servings (minimum: 1)
</ParamField>

<ParamField body="ingredients" type="Ingredient[]">
  Array of ingredient objects. This will replace the entire ingredients array.
  
  <ParamField body="name" type="string" required>
    Ingredient name
  </ParamField>
  
  <ParamField body="amount" type="number" required>
    Quantity of the ingredient (must be positive)
  </ParamField>
  
  <ParamField body="unit" type="string" required>
    Unit of measurement: `g`, `ml`, or `unit`
  </ParamField>
  
  <ParamField body="nutritionBasis" type="object" required>
    Nutrition information basis
    
    <ParamField body="amount" type="number" required>
      Basis amount for nutrition calculation (must be positive)
    </ParamField>
    
    <ParamField body="unit" type="string" required>
      Basis unit: `g`, `ml`, or `unit`
    </ParamField>
  </ParamField>
  
  <ParamField body="macrosPerBasis" type="object" required>
    Macronutrients per nutrition basis
    
    <ParamField body="kcal" type="number" required>
      Calories (kcal) - non-negative
    </ParamField>
    
    <ParamField body="protein" type="number" required>
      Protein in grams - non-negative
    </ParamField>
    
    <ParamField body="carbohydrate" type="number" required>
      Carbohydrates in grams - non-negative
    </ParamField>
    
    <ParamField body="fat" type="number" required>
      Fat in grams - non-negative
    </ParamField>
  </ParamField>
  
  <ParamField body="gramsPerUnit" type="number">
    Weight in grams per unit (optional, for unit-based ingredients). Must be positive.
  </ParamField>
  
  <ParamField body="densityGPerMl" type="number">
    Density in grams per milliliter (optional, for liquid ingredients). Must be positive.
  </ParamField>
</ParamField>

<ParamField body="categoryIds" type="string[]">
  Array of category IDs. This will replace the entire categoryIds array.
</ParamField>

## Response

<ResponseField name="_id" type="string">
  Unique recipe ID
</ResponseField>

<ResponseField name="userId" type="string">
  ID of the user who owns this recipe
</ResponseField>

<ResponseField name="name" type="string">
  Recipe name
</ResponseField>

<ResponseField name="date" type="string">
  Date in YYYY-MM-DD format
</ResponseField>

<ResponseField name="servings" type="integer">
  Number of servings
</ResponseField>

<ResponseField name="ingredients" type="Ingredient[]">
  Array of ingredient objects (see request body for structure)
</ResponseField>

<ResponseField name="categoryIds" type="string[]">
  Array of category IDs
</ResponseField>

<ResponseField name="createdAt" type="string">
  Timestamp when the recipe was created
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Timestamp when the recipe was last updated
</ResponseField>
