---
title: "List Recipes"
api: "GET /api/v1/recipes"
description: "Retrieve a paginated list of recipes for the authenticated user"
---

Returns a paginated list of recipes. Supports filtering by search query and category, as well as custom sorting.

## Query Parameters

<ParamField query="page" type="integer" default="1">
  Page number for pagination (minimum: 1)
</ParamField>

<ParamField query="limit" type="integer" default="20">
  Number of items per page (minimum: 1, maximum: 100)
</ParamField>

<ParamField query="q" type="string">
  Optional search query to filter recipes by name (case-insensitive)
</ParamField>

<ParamField query="category" type="string">
  Filter recipes by category ID
</ParamField>

<ParamField query="sort" type="string" default="-createdAt">
  Sort order for results. Options:
  - `name` - Sort by name (A-Z)
  - `-name` - Sort by name (Z-A)
  - `createdAt` - Sort by creation date (oldest first)
  - `-createdAt` - Sort by creation date (newest first)
</ParamField>

## Response

<ResponseField name="items" type="Recipe[]">
  Array of recipe objects
  
  <ResponseField name="_id" type="string">
    Recipe ID
  </ResponseField>
  
  <ResponseField name="userId" type="string">
    ID of the user who owns this recipe
  </ResponseField>
  
  <ResponseField name="name" type="string">
    Recipe name
  </ResponseField>
  
  <ResponseField name="date" type="string">
    Date in YYYY-MM-DD format
  </ResponseField>
  
  <ResponseField name="servings" type="integer" default="1">
    Number of servings (minimum: 1)
  </ResponseField>
  
  <ResponseField name="ingredients" type="Ingredient[]">
    Array of ingredient objects
    
    <ResponseField name="name" type="string">
      Ingredient name
    </ResponseField>
    
    <ResponseField name="amount" type="number">
      Quantity of the ingredient
    </ResponseField>
    
    <ResponseField name="unit" type="string">
      Unit of measurement: `g`, `ml`, or `unit`
    </ResponseField>
    
    <ResponseField name="nutritionBasis" type="object">
      Nutrition information basis
      
      <ResponseField name="amount" type="number">
        Basis amount for nutrition calculation
      </ResponseField>
      
      <ResponseField name="unit" type="string">
        Basis unit: `g`, `ml`, or `unit`
      </ResponseField>
    </ResponseField>
    
    <ResponseField name="macrosPerBasis" type="object">
      Macronutrients per nutrition basis
      
      <ResponseField name="kcal" type="number">
        Calories (kcal)
      </ResponseField>
      
      <ResponseField name="protein" type="number">
        Protein in grams
      </ResponseField>
      
      <ResponseField name="carbohydrate" type="number">
        Carbohydrates in grams
      </ResponseField>
      
      <ResponseField name="fat" type="number">
        Fat in grams
      </ResponseField>
    </ResponseField>
    
    <ResponseField name="gramsPerUnit" type="number">
      Weight in grams per unit (optional, for unit-based ingredients)
    </ResponseField>
    
    <ResponseField name="densityGPerMl" type="number">
      Density in grams per milliliter (optional, for liquid ingredients)
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="categoryIds" type="string[]">
    Array of category IDs associated with this recipe
  </ResponseField>
  
  <ResponseField name="createdAt" type="string">
    Timestamp when the recipe was created
  </ResponseField>
  
  <ResponseField name="updatedAt" type="string">
    Timestamp when the recipe was last updated
  </ResponseField>
</ResponseField>

<ResponseField name="page" type="integer">
  Current page number
</ResponseField>

<ResponseField name="limit" type="integer">
  Number of items per page
</ResponseField>

<ResponseField name="total" type="integer">
  Total number of recipes matching the filter
</ResponseField>

<RequestExample>
```bash cURL
curl -X GET "https://api.fitboard.app/api/v1/recipes?page=1&limit=20&sort=-createdAt" \
  -H "Authorization: Bearer YOUR_TOKEN"
```

```javascript JavaScript
const response = await fetch('https://api.fitboard.app/api/v1/recipes?page=1&limit=20', {
  headers: {
    'Authorization': 'Bearer YOUR_TOKEN'
  }
});

const data = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://api.fitboard.app/api/v1/recipes',
    params={'page': 1, 'limit': 20, 'sort': '-createdAt'},
    headers={'Authorization': 'Bearer YOUR_TOKEN'}
)

data = response.json()
```
